1142TensorErrorCode matrix_float_lu(MatrixFloatHandle handle, MatrixFloatHandle* out_L, MatrixFloatHandle* out_U, size_t** out_pivot, size_t* out_pivot_size) {
1143 if (!handle || !out_L || !out_U || !out_pivot || !out_pivot_size) return TENSOR_ERROR_NULL_POINTER;
1144
1145 TENSOR_TRY_BEGIN
1146 auto* mat = static_cast<Matrixf*>(handle);
1147 auto result = linalg::lu_decomp(*mat);
1148
1149 if (auto* lu_data = std::get_if<std::pair<Matrixf, std::vector<int>>>(&result)) {
1150 const Matrixf& LU = lu_data->first;
1151 const std::vector<int>& pivots = lu_data->second;
1152
1153 auto dims = LU.dims();
1154 size_t m = dims[0];
1155 size_t n = dims[1];
1156 size_t min_dim = std::min(m, n);
1157
1158 // Use smart pointers for exception safety
1159 std::unique_ptr<Matrixf> L(new Matrixf({m, min_dim}));
1160 for (size_t i = 0; i < m; ++i) {
1161 for (size_t j = 0; j < min_dim; ++j) {
1162 if (i > j) {
1163 (*L)[{i, j}] = LU[{i, j}];
1164 } else if (i == j) {
1165 (*L)[{i, j}] = 1.0f;
1166 } else {
1167 (*L)[{i, j}] = 0.0f;
1168 }
1169 }
1170 }
1171
1172 std::unique_ptr<Matrixf> U(new Matrixf({min_dim, n}));
1173 for (size_t i = 0; i < min_dim; ++i) {
1174 for (size_t j = 0; j < n; ++j) {
1175 if (i <= j) {
1176 (*U)[{i, j}] = LU[{i, j}];
1177 } else {
1178 (*U)[{i, j}] = 0.0f;
1179 }
1180 }
1181 }

1186 static_cast<size_t*>(malloc(pivots.size() * sizeof(size_t)))
1187 );

1189 return TENSOR_ERROR_ALLOCATION;

1191 for (size_t i = 0; i < pivots.size(); ++i) {
1192 pivot_copy[i] = static_cast<size_t>(pivots[i]);
1193 }
1194
1195 // Transfer ownership to C API
1196 *out_L = L.release();
1197 *out_U = U.release();
1198 *out_pivot = pivot_copy.release();
1199 *out_pivot_size = pivots.size();

1202 }
