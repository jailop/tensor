144template<typename T, size_t N>
145class SGD : public Optimizer<T> {
146private:
147 std::vector<Tensor<T, N>*> parameters_;
148 std::vector<Tensor<T, N>> velocities_; // For momentum
149 T momentum_;
150 T weight_decay_;
151 bool use_momentum_;
152
153public:
161 SGD(std::vector<Tensor<T, N>*> parameters, T learning_rate,
162 T momentum = T(0), T weight_decay = T(0))
163 : Optimizer<T>(learning_rate),
164 parameters_(parameters),
165 momentum_(momentum),
166 weight_decay_(weight_decay),
167 use_momentum_(momentum > T(0)) {
168
169 // Initialize velocity buffers for momentum
170 if (use_momentum_) {
171 for (auto* param : parameters_) {
172 velocities_.emplace_back(param->dims(), param->uses_gpu(), false);
173 velocities_.back().fill(T(0));
174 }
175 }
176 }
177
178 void step() override {
179 for (size_t i = 0; i < parameters_.size(); ++i) {
180 auto* param = parameters_[i];
181
182 if (!param->requires_grad() || !param->grad()) {
183 continue;
184 }
185
186 auto* grad = param->grad();
187
188 // Apply weight decay if specified
189 if (weight_decay_ > T(0)) {
190 size_t total = param->total_size();
191 for (size_t j = 0; j < total; ++j) {
192 grad->data()[j] += weight_decay_ * param->data()[j];
193 }
194 }
195
196 if (use_momentum_) {
197 // Update velocity: v = momentum * v + lr * grad
198 auto& velocity = velocities_[i];
199 size_t total = param->total_size();
200
201 for (size_t j = 0; j < total; ++j) {
202 velocity.data()[j] = momentum_ * velocity.data()[j] +
203 this->learning_rate_ * grad->data()[j];
204 param->data()[j] -= velocity.data()[j];
205 }
206 } else {
207 // Standard SGD: param = param - lr * grad
208 size_t total = param->total_size();
209 for (size_t j = 0; j < total; ++j) {
210 param->data()[j] -= this->learning_rate_ * grad->data()[j];
211 }
212 }
213 }
214 }
215
216 void zero_grad() override {
217 for (auto* param : parameters_) {
218 if (param->grad()) {
219 param->zero_grad();
220 }
221 }
222 }
223};

238template<typename T, size_t N>
239class Adam : public Optimizer<T> {
240private:
241 std::vector<Tensor<T, N>*> parameters_;
242 std::vector<Tensor<T, N>> first_moments_; // m_t (gradient moving average)
243 std::vector<Tensor<T, N>> second_moments_; // v_t (squared gradient moving average)
244 T beta1_;
245 T beta2_;
246 T epsilon_;
247 T weight_decay_;
248 size_t step_count_;
249
250public:
260 Adam(std::vector<Tensor<T, N>*> parameters,
261 T learning_rate = T(0.001),
262 T beta1 = T(0.9),
263 T beta2 = T(0.999),
264 T epsilon = T(1e-8),
265 T weight_decay = T(0))
266 : Optimizer<T>(learning_rate),
267 parameters_(parameters),
268 beta1_(beta1),
269 beta2_(beta2),
270 epsilon_(epsilon),
271 weight_decay_(weight_decay),
272 step_count_(0) {
273
274 // Initialize moment buffers
275 for (auto* param : parameters_) {
276 first_moments_.emplace_back(param->dims(), param->uses_gpu(), false);
277 first_moments_.back().fill(T(0));
278
279 second_moments_.emplace_back(param->dims(), param->uses_gpu(), false);
280 second_moments_.back().fill(T(0));
281 }
282 }
283
284 void step() override {
285 step_count_++;
286
287 // Compute bias correction terms
288 T bias_correction1 = T(1) - std::pow(beta1_, static_cast<T>(step_count_));
289 T bias_correction2 = T(1) - std::pow(beta2_, static_cast<T>(step_count_));
290
291 for (size_t i = 0; i < parameters_.size(); ++i) {
292 auto* param = parameters_[i];
293
294 if (!param->requires_grad() || !param->grad()) {
295 continue;
296 }
297
298 auto* grad = param->grad();
299 auto& m = first_moments_[i];
300 auto& v = second_moments_[i];
301
302 size_t total = param->total_size();
303
304 for (size_t j = 0; j < total; ++j) {
305 T grad_val = grad->data()[j];
306
307 // Apply weight decay
308 if (weight_decay_ > T(0)) {
309 grad_val += weight_decay_ * param->data()[j];
310 }
311
312 // Update biased first moment estimate
313 m.data()[j] = beta1_ * m.data()[j] + (T(1) - beta1_) * grad_val;
314
315 // Update biased second moment estimate
316 v.data()[j] = beta2_ * v.data()[j] + (T(1) - beta2_) * grad_val * grad_val;
317
318 // Compute bias-corrected moment estimates
319 T m_hat = m.data()[j] / bias_correction1;
320 T v_hat = v.data()[j] / bias_correction2;
321
322 // Update parameters
323 param->data()[j] -= this->learning_rate_ * m_hat / (std::sqrt(v_hat) + epsilon_);
324 }
325 }
326 }
327
328 void zero_grad() override {
329 for (auto* param : parameters_) {
330 if (param->grad()) {
331 param->zero_grad();
332 }
333 }
334 }
335
339 void reset() {
340 step_count_ = 0;
341 for (auto& m : first_moments_) {
342 m.fill(T(0));
343 }
344 for (auto& v : second_moments_) {
345 v.fill(T(0));
346 }
347 }
348};

360template<typename T, size_t N>
361class RMSprop : public Optimizer<T> {
362private:
363 std::vector<Tensor<T, N>*> parameters_;
364 std::vector<Tensor<T, N>> square_avg_; // Running average of squared gradients
365 T alpha_; // Smoothing constant
366 T epsilon_; // Small constant for numerical stability
367 T weight_decay_;
368 T momentum_;
369 std::vector<Tensor<T, N>> momentum_buffer_;
370 bool use_momentum_;
371
372public:
382 RMSprop(std::vector<Tensor<T, N>*> parameters,
383 T learning_rate = T(0.01),
384 T alpha = T(0.99),
385 T epsilon = T(1e-8),
386 T weight_decay = T(0),
387 T momentum = T(0))
388 : Optimizer<T>(learning_rate),
389 parameters_(parameters),
390 alpha_(alpha),
391 epsilon_(epsilon),
392 weight_decay_(weight_decay),
393 momentum_(momentum),
394 use_momentum_(momentum > T(0)) {
395
396 // Initialize buffers
397 for (auto* param : parameters_) {
398 square_avg_.emplace_back(param->dims(), param->uses_gpu(), false);
399 square_avg_.back().fill(T(0));
400
401 if (use_momentum_) {
402 momentum_buffer_.emplace_back(param->dims(), param->uses_gpu(), false);
403 momentum_buffer_.back().fill(T(0));
404 }
405 }
406 }
407
408 void step() override {
409 for (size_t i = 0; i < parameters_.size(); ++i) {
410 auto* param = parameters_[i];
411
412 if (!param->requires_grad() || !param->grad()) {
413 continue;
414 }
415
416 auto* grad = param->grad();
417 auto& v = square_avg_[i];
418
419 size_t total = param->total_size();
420
421 for (size_t j = 0; j < total; ++j) {
422 T grad_val = grad->data()[j];
423
424 // Apply weight decay
425 if (weight_decay_ > T(0)) {
426 grad_val += weight_decay_ * param->data()[j];
427 }
428
429 // Update moving average of squared gradient
430 v.data()[j] = alpha_ * v.data()[j] + (T(1) - alpha_) * grad_val * grad_val;
431
432 if (use_momentum_) {
433 auto& buf = momentum_buffer_[i];
434 buf.data()[j] = momentum_ * buf.data()[j] +
435 this->learning_rate_ * grad_val / (std::sqrt(v.data()[j]) + epsilon_);
436 param->data()[j] -= buf.data()[j];
437 } else {
438 // Update parameters
439 param->data()[j] -= this->learning_rate_ * grad_val / (std::sqrt(v.data()[j]) + epsilon_);
440 }
441 }
442 }
443 }
444
445 void zero_grad() override {
446 for (auto* param : parameters_) {
447 if (param->grad()) {
448 param->zero_grad();
449 }
450 }
451 }
