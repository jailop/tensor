50template<typename T>
51class MemoryPool {
52private:
53 struct Block {
54 T* data;
55 size_t size;
56
57 Block(T* d, size_t s) : data(d), size(s) {}
58 };
59
60 std::vector<Block> free_blocks_;
61 std::mutex mutex_;
62 size_t total_allocated_ = 0;
63 size_t total_freed_ = 0;
64
65public:
69 MemoryPool() = default;
70
74 ~MemoryPool() {
75 std::lock_guard<std::mutex> lock(mutex_);
76 for (auto& block : free_blocks_) {
77 delete[] block.data;
78 }
79 }
80
89 T* allocate(size_t size) {
90 std::lock_guard<std::mutex> lock(mutex_);
91
92 // Try to find a suitable block from the pool
93 for (auto it = free_blocks_.begin(); it != free_blocks_.end(); ++it) {
94 if (it->size >= size && it->size < size * 2) {
95 T* data = it->data;
96 free_blocks_.erase(it);
97 total_allocated_++;
98 return data;
99 }
100 }
101
102 // No suitable block found, allocate new
103 total_allocated_++;
104 return new T[size];
105 }
106
114 void deallocate(T* ptr, size_t size) {
115 if (!ptr) return;
116
117 std::lock_guard<std::mutex> lock(mutex_);
118 total_freed_++;
119
120 // Keep pool size reasonable - limit to 100 blocks
121 if (free_blocks_.size() < 100) {
122 free_blocks_.emplace_back(ptr, size);
123 } else {
124 delete[] ptr;
125 }
126 }
127
133 void clear() {
134 std::lock_guard<std::mutex> lock(mutex_);
135 for (auto& block : free_blocks_) {
136 delete[] block.data;
137 }
138 free_blocks_.clear();
139 }
140
145 std::pair<size_t, size_t> stats() const {
146 std::lock_guard<std::mutex> lock(const_cast<std::mutex&>(mutex_));
147 return {total_allocated_, total_freed_};
148 }
149};

181class ThreadPool {
182private:
183 std::vector<std::thread> workers_;
184 std::queue<std::function<void()>> tasks_;
185 std::mutex queue_mutex_;
186 std::condition_variable condition_;
187 bool stop_ = false;
188
189public:
194 explicit ThreadPool(size_t num_threads = std::thread::hardware_concurrency()) {
195 for (size_t i = 0; i < num_threads; ++i) {
196 workers_.emplace_back([this] {
197 while (true) {
198 std::function<void()> task;
199 {
200 std::unique_lock<std::mutex> lock(queue_mutex_);
201 condition_.wait(lock, [this] { return stop_ || !tasks_.empty(); });
202
203 if (stop_ && tasks_.empty()) {
204 return;
205 }
206
207 task = std::move(tasks_.front());
208 tasks_.pop();
209 }
210 task();
211 }
212 });
213 }
214 }
215
219 ~ThreadPool() {
220 {
221 std::unique_lock<std::mutex> lock(queue_mutex_);
222 stop_ = true;
223 }
224 condition_.notify_all();
225 for (auto& worker : workers_) {
226 if (worker.joinable()) {
227 worker.join();
228 }
229 }
230 }
231
240 template<typename F, typename... Args>
241 auto enqueue(F&& f, Args&&... args)
242 -> std::future<typename std::invoke_result<F, Args...>::type> {
243 using return_type = typename std::invoke_result<F, Args...>::type;
244
245 auto task = std::make_shared<std::packaged_task<return_type()>>(
246 std::bind(std::forward<F>(f), std::forward<Args>(args)...)
247 );
248
249 std::future<return_type> res = task->get_future();
250 {
251 std::unique_lock<std::mutex> lock(queue_mutex_);
252 if (stop_) {
253 throw std::runtime_error("Enqueue on stopped ThreadPool");
254 }
255 tasks_.emplace([task]() { (*task)(); });
256 }
257 condition_.notify_one();
258 return res;
259 }
260
265 size_t size() const {
266 return workers_.size();
267 }
268};

294inline void parallel_for(size_t start, size_t end,
295 std::function<void(size_t)> func,
296 size_t min_per_thread = 1000) {
297 size_t range = end - start;
298 if (range < min_per_thread) {
299 // Not worth parallelizing
300 for (size_t i = start; i < end; ++i) {
301 func(i);
302 }
303 return;
304 }
305
306 auto& pool = get_thread_pool();
307 size_t num_threads = std::min(range / min_per_thread, pool.size());
308
309 if (num_threads <= 1) {
310 for (size_t i = start; i < end; ++i) {
311 func(i);
312 }
313 return;
314 }
315
316 size_t chunk_size = range / num_threads;
317 std::vector<std::future<void>> futures;
318
319 for (size_t t = 0; t < num_threads; ++t) {
320 size_t chunk_start = start + t * chunk_size;
321 size_t chunk_end = (t == num_threads - 1) ? end : chunk_start + chunk_size;
322
323 futures.push_back(pool.enqueue([chunk_start, chunk_end, &func]() {
324 for (size_t i = chunk_start; i < chunk_end; ++i) {
325 func(i);
326 }
327 }));
328 }
329
330 for (auto& f : futures) {
331 f.get();
332 }
333}

347struct Float16 {
348 uint16_t data;
349
350 Float16() : data(0) {}
351
356 explicit Float16(float value) {
357 uint32_t f;
358 std::memcpy(&f, &value, sizeof(float));
359
360 uint32_t sign = (f >> 16) & 0x8000;
361 int32_t exponent = ((f >> 23) & 0xFF) - 127 + 15;
362 uint32_t mantissa = f & 0x7FFFFF;
363
364 if (exponent <= 0) {
365 // Subnormal or zero
366 data = sign;
367 } else if (exponent >= 31) {
368 // Infinity or overflow
369 data = sign | 0x7C00;
370 } else {
371 // Normal number
372 data = sign | (exponent << 10) | (mantissa >> 13);
373 }
374 }
375
380 float to_float() const {
381 uint32_t sign = (data & 0x8000) << 16;
382 uint32_t exponent = (data & 0x7C00) >> 10;
383 uint32_t mantissa = (data & 0x3FF) << 13;
384
385 uint32_t f;
386 if (exponent == 0) {
387 // Subnormal or zero
388 f = sign;
389 } else if (exponent == 31) {
390 // Infinity
391 f = sign | 0x7F800000 | mantissa;
392 } else {
393 // Normal number
394 f = sign | ((exponent - 15 + 127) << 23) | mantissa;
395 }
396
397 float result;
398 std::memcpy(&result, &f, sizeof(float));
399 return result;
400 }
401
402 operator float() const { return to_float(); }
403};

422 explicit BFloat16(float value) {
423 uint32_t f;
424 std::memcpy(&f, &value, sizeof(float));
425
426 // BF16 is simply the top 16 bits of FP32

428 uint32_t rounding = (f >> 16) & 1; // Round to nearest even
429 data = (f >> 16) + rounding;

437 uint32_t f = static_cast<uint32_t>(data) << 16;
438 float result;
439 std::memcpy(&result, &f, sizeof(float));

441 }

573template<typename T>
574std::function<void(T*, const T*, size_t)>
575fuse_operations(const std::vector<OperationType>& ops) {
576 return [ops](T* dst, const T* src, size_t size) {
577 for (size_t i = 0; i < size; ++i) {
578 T val = src[i];
579 for (auto op : ops) {
580 switch (op) {
581 case OperationType::Exp:
582 val = std::exp(val);
583 break;
584 case OperationType::Tanh:
585 val = std::tanh(val);
586 break;
587 case OperationType::Sigmoid:
588 val = T(1) / (T(1) + std::exp(-val));
589 break;
590 case OperationType::ReLU:
591 val = std::max(T(0), val);
592 break;
593 default:
594 break;
595 }
596 }
597 dst[i] = val;
598 }
599 };
